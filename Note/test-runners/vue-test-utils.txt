it('should open create company dialog box when create icon clicked', async () => {
    // Arrange
    const createCompanyButton = wrapper.find(selectors.createCompanyButton);
    // Assert
    expect(createCompanyButton.exists()).to.be.equal(true);
    // Acts
    await createCompanyButton.trigger('click');
    // Assertion
    expect(wrapper.vm.isCreateCompanyModal).to.be.equal(true);
    // Arrange
    const createCompanyModal = wrapper.find(selectors.createCompanyModal);
    // Assertion
    expect(createCompanyModal.isVisible()).to.be.equal(true);
});

# Terminologies:
    - test suite: describe()
    - test case: it()
    - instanciate component to be tested: mount() ---> wrapper contains all the props,methods of a corresponding component(proxy of component)
    - assertion-statement: expect()
    - assertion-check: .toBe()

    > expect(wrapper.text).toBe("I am Simba.")
    > expect(Received).toBe(Expected)
